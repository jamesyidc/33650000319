  • й—®йўҳеңәжҷҜпјҡRSIй«ҳдҪҚпјҲеҰӮ1190пјүж—¶ratioж»Ўи¶ідҪҶдёҚжҳҜзңҹеә•йғЁ

1пёҸвғЈ B-A-Cжіўеі°жЈҖжөӢзі»з»ҹ

жҷәиғҪиҜҶеҲ«еёӮеңәжіўеі°жіўи°·пјҢзІҫеҮҶжҚ•жҚүд№°еҚ–ж—¶жңәпјҡ

  • BзӮ№пјҲи°·еә•пјүпјҡи“қиүІеңҶеҪўж Үи®°пјҢеёӮеңәжңҖдҪҺзӮ№пјҢжҪңеңЁд№°е…Ҙжңәдјҡ
  • AзӮ№пјҲеі°йЎ¶пјүпјҡж©ҷиүІдёүи§’ж Үи®°пјҢеёӮеңәжңҖй«ҳзӮ№пјҢжҪңеңЁеҚ–еҮәжңәдјҡ
  • CзӮ№пјҲеӣһи°ғпјүпјҡзҙ«иүІиҸұеҪўж Үи®°пјҢеҸҚеј№еҗҺзҡ„еӣһи°ғзӮ№пјҢи§ӮеҜҹж”Ҝж’‘дҪҚ
  • жҢҜе№…и®Ўз®—пјҡиҮӘеҠЁи®Ўз®—BеҲ°Aзҡ„ж¶Ёе№…е’ҢAеҲ°Cзҡ„еӣһж’Өе№…еәҰ
  • еҺҶеҸІжҹҘзңӢпјҡж”ҜжҢҒжҹҘзңӢеҺҶеҸІж—Ҙжңҹзҡ„жіўеі°ж Үи®°пјҢеҲҶжһҗеҺҶеҸІеҪўжҖҒ
  • еҸҜи§ҶеҢ–еўһејәпјҡеӨ§еҸ·ж Үи®°зӮ№пјҲ18-20pxпјү+ зІ—иҫ№жЎҶпјҲ3pxпјү+ й«ҳеҜ№жҜ”еәҰеӯ—дҪ“
  • жҷәиғҪйҳҲеҖјпјҡжҢҜе№…вүҘ35%гҖҒзӘ—еҸЈжңҹ15еҲҶй’ҹпјҢиҝҮж»Өе°ҸжіўеҠЁ

рҹ“Ҳ жіўеі°еҪўжҖҒи§ЈиҜ»пјҡ

еҪўжҖҒ зү№еҫҒ дәӨжҳ“еҗ«д№ү
BзӮ№еҮәзҺ° д»·ж ји§Ұеә•еҸҚеј№ рҹҹў иҖғиҷ‘д№°е…ҘпјҢи®ҫзҪ®жӯўжҚҹ
AзӮ№еҮәзҺ° д»·ж ји§ҒйЎ¶еӣһиҗҪ рҹ”ҙ иҖғиҷ‘еҚ–еҮәпјҢй”Ғе®ҡеҲ©ж¶Ұ
CзӮ№еҮәзҺ° еӣһи°ғжөӢиҜ•ж”Ҝж’‘ рҹҹЎ и§ӮеҜҹжҳҜеҗҰи·Ңз ҙпјҢеҶіе®ҡиҝӣеҮә
CзӮ№е®ҲдҪҸ ж”Ҝж’‘жңүж•Ҳ вң… еҸҜиғҪз»§з»ӯдёҠж¶ЁпјҢжҢҒжңүжҲ–еҠ д»“
CзӮ№и·Ңз ҙ ж”Ҝж’‘еӨұж•Ҳ вҡ пёҸ и¶ӢеҠҝиҪ¬ејұпјҢиҖғиҷ‘зҰ»еңә

2пёҸвғЈ еҒҮзӘҒз ҙдҝЎеҸ·жЈҖжөӢ

иҮӘеҠЁиҜҶеҲ«иҷҡеҒҮзӘҒз ҙпјҢйҒҝе…ҚиҝҪй«ҳиў«еҘ—пјҡ

  • жЈҖжөӢжқЎд»¶пјҡиҝһз»ӯ3дёӘжіўеі°зҡ„AзӮ№еқҮжңӘзӘҒз ҙз¬¬1дёӘжіўеі°зҡ„еүҚй«ҳ
  • еёӮеңәеҗ«д№үпјҡдёҠж¶ЁеҠЁиғҪеҮҸејұпјҢй«ҳдҪҚзӣҳж•ҙпјҢеҸҜиғҪиҪ¬дёәдёӢи·Ңи¶ӢеҠҝ
  • йў„иӯҰеұ•зӨәпјҡйЎөйқўйЎ¶йғЁзәўиүІйў„иӯҰжЎҶпјҢжҳҫзӨәAзӮ№еҜ№жҜ”е’ҢзӘҒз ҙеҲӨж–ӯ
  • иҜҰз»ҶеҲҶжһҗпјҡеұ•зӨәA1гҖҒA2гҖҒA3зҡ„е…·дҪ“ж•°еҖје’Ңзӣёдә’еҜ№жҜ”
  • дәӨжҳ“е»әи®®пјҡж №жҚ®е…·дҪ“жғ…еҶөжҸҗдҫӣзҰ»еңәжҲ–и§Ӯжңӣе»әи®®

рҹҺҜ еҒҮзӘҒз ҙжЎҲдҫӢпјҡ

  • жіўеі°1пјҡA1 = +40%пјҲйҰ–ж¬ЎеҶІй«ҳпјү
  • жіўеі°2пјҡA2 = +35%пјҲжңӘзӘҒз ҙ40%пјҢз¬¬дёҖж¬ЎеҒҮзӘҒз ҙпјү
  • жіўеі°3пјҡA3 = +32%пјҲжңӘзӘҒз ҙ40%пјҢз¬¬дәҢж¬ЎеҒҮзӘҒз ҙпјү
  • з»“и®әпјҡиҝһз»ӯдёӨж¬ЎеҶІй«ҳеӨұиҙҘпјҢдёҠж¶ЁеҠЁиғҪиЎ°з«ӯпјҢеёӮеңәеҸҜиғҪиҪ¬и·Ң вҡ пёҸ

3пёҸвғЈ жҡҙи·Ңйў„иӯҰзі»з»ҹпјҲAX < AX+1 < AX+2пјү

жЈҖжөӢиҝһз»ӯеҸҚеј№й«ҳзӮ№йҖ’еўһжЁЎејҸпјҢйў„иӯҰжҪңеңЁжҡҙи·ҢйЈҺйҷ©пјҡ

  • AзӮ№йҖ’еўһпјҡжЈҖжөӢA1 < A2 < A3зҡ„еҸҚеј№й«ҳзӮ№йҖ’еўһи¶ӢеҠҝ
  • еёӮеңәеҗ«д№үпјҡеҸҚеј№й«ҳзӮ№еҚҮй«ҳпјҢдҪҶе®һйҷ…жҳҜдёӢи·Ңи¶ӢеҠҝдёӯзҡ„еҸҚеј№пјҢжҡҙи·ҢеүҚе…Ҷ
  • BзӮ№йҖ’еҮҸпјҡеҗҢж—¶жЈҖжөӢB1 > B2 > B3пјҢи°·еә•и¶ҠжқҘи¶ҠдҪҺпјҢйЈҺйҷ©жӣҙй«ҳ
  • йЈҺйҷ©зӯүзә§пјҡAйҖ’еўһ=дёӯйЈҺйҷ©пјҢAйҖ’еўһ+BйҖ’еҮҸ=й«ҳйЈҺйҷ©
  • йў„иӯҰеұ•зӨәпјҡж©ҷиүІ/зәўиүІйў„иӯҰжЎҶпјҢеұ•зӨәAзӮ№е’ҢBзӮ№и¶ӢеҠҝеҲҶжһҗ
  • дәӨжҳ“е»әи®®пјҡй«ҳйЈҺйҷ©вҶ’з«ӢеҚізҰ»еңәжҲ–ејҖз©әпјӣдёӯйЈҺйҷ©вҶ’и°Ёж…Һи§Ӯжңӣ

вҡ пёҸ жҡҙи·Ңйў„иӯҰжЎҲдҫӢпјҡ

  • жңҖдҪҺзӮ№пјҡз»ҝиүІеҗ‘дёӢз®ӯеӨҙж Үи®°пјҢжҳҫзӨәеҪ“ж—ҘRSIд№Ӣе’ҢжңҖе°ҸеҖј

  • е®һж—¶жӣҙж–°пјҡйҡҸж•°жҚ®жӣҙж–°иҮӘеҠЁеҲ·ж–°пјҢдҝқжҢҒеҗҢжӯҘ

import requests, json
from datetime import datetime, timezone, timedelta

symbols = ["BTC", "ETH", "BNB", "XRP", "DOGE", "SOL", "DOT", "LTC", "LINK", 
           "HBAR", "TAO", "CFX", "TRX", "TON", "NEAR", "LDO", "CRO", "ETC", 
           "XLM", "BCH", "UNI", "SUI", "FIL", "STX", "CRV", "AAVE", "APT"]

baseline = {}
for symbol in symbols:
    try:
        url = f"https://www.okx.com/api/v5/market/candles?instId={symbol}-USDT-SWAP&bar=1D&limit=2"
        resp = requests.get(url, timeout=5)
        data = resp.json()
        if data['code'] == '0' and len(data['data']) > 0:
            # data[0][1] жҳҜд»ҠеӨ©зҡ„ејҖзӣҳд»·
            open_price = float(data['data'][0][1])
            baseline[symbol] = open_price
            print(f"{symbol}: {open_price}")
    except:
        pass
